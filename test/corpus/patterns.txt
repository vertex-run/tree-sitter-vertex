fn describe(s: Shape): String {
    match s {
        Circle(radius: r) => "circle",
        Rectangle(width: w, height: h) => "rect",
        Triangle => "tri",
    }
}

        Saturday | Sunday => true,
        _ => false,

fn tail(xs: List<Int>): List<Int> {
    match xs {
        [_, ...rest] => rest,
        [h, ..] => [h],
        _ => [],
    }
}
